Exercice 6 : Créer un sélecteur de couleur

Objectif : Changer la couleur de fond de la page avec des boîtes interactives

CODE JAVASCRIPT
function changerFond(couleur) {
    
    document.body.style.backgroundColor = couleur;
   
    document.querySelectorAll(".boite").forEach(box => 
        box.classList.remove("actif"));
   
    event.currentTarget.classList.add("actif");
}
function reinitialiserFond() {
    
    document.body.style.backgroundColor = "var(--bg-primary)";
    
    document.querySelectorAll(".boite").forEach(box => 
        box.classList.remove("actif"));
}
