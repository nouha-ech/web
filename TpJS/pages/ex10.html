Objectif : Créer un système d'onglets avec affichage/masquage dynamique

function afficherOnglet(numero) {
    document.querySelectorAll(".contenu-onglet").forEach(content => {
        content.classList.remove("visible");
        content.classList.add("cache");
    });
    const contenuSelect = document.querySelector("#contenu" + numero);
    contenuSelect.classList.remove("cache");
    contenuSelect.classList.add("visible");
    document.querySelectorAll(".onglet").forEach(btn => 
        btn.classList.remove("actif"));
    event.currentTarget.classList.add("actif");
}
