Exercice 9 : Compteur avec conditions

Objectif : Compteur qui change de couleur selon la valeur

CODE JAVASCRIPT
let compteurValue = 0;

function ajusterCouleur(valeur) {
    const compteur = document.querySelector("#compteur");
    if (valeur < 0) {
        compteur.style.color = "#e57373"; // Rouge si négatif
    } else if (valeur > 0) {
        compteur.style.color = "#81c784"; // Vert si positif
    } else {
        compteur.style.color = "var(--text-primary)"; // Noir si zéro
    }
}
function incrementer() {
    compteurValue++;
    document.querySelector("#compteur").innerText = compteurValue;
    ajusterCouleur(compteurValue);
}
function decrementer() {
    compteurValue--;
    document.querySelector("#compteur").innerText = compteurValue;
    ajusterCouleur(compteurValue);
}
function reinitialiser() {
    compteurValue = 0;
    document.querySelector("#compteur").innerText = compteurValue;
    ajusterCouleur(compteurValue);
}
APERÇU

0
