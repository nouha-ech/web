Exercice 5 : Modifier plusieurs éléments

Objectif : Sélectionner et modifier tous les éléments d'une classe

CODE JAVASCRIPT
function colorierTous() {
    
    document.querySelectorAll(".item").forEach(item => {
        item.classList.add("fond-jaune");
    });
}
function numeroteTous() {

    document.querySelectorAll(".item").forEach((item, index) => {
        const texteOriginal = item.innerText.replace(/^[0-9]+\.\s*/, '');
        item.innerText = (index + 1) + ". " + texteOriginal;
    });
}
