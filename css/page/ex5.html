5.1 Mobile First + Media Queries

CODE CSS
/* Mobile First */
.container {
  width: 100%;
  padding: 15px;
}

.nav-menu {
  display: flex;
  flex-direction: column;
  gap: 5px;
}

.content {
  display: block;
}

.sidebar {
  display: none;
}

/* Tablette */
@media (min-width: 768px) {
  .nav-menu {
    flex-direction: row;
  }

  .content {
    display: flex;
  }

  .sidebar {
    display: block;
    width: 30%;
  }

  .main {
    width: 70%;
  }
}

/* Desktop */
@media (min-width: 1024px) {
  .sidebar {
    width: 25%;
  }

  .main {
    width: 75%;
  }
}
